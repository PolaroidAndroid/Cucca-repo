###anova
trat<-read.table("trattamenti.txt",header=T) #lettura dati
#i valori vanno impilati e i nomi di colonna diventano etichette di trattamento
trats<-stack(trat)
#analisi della varianza
anova(aov(trats[,1]~trats[,2]))
